The binary is very small. Running the size command on the binary shows us that the .text section is a mere 446 bytes and the .data section is just 117 bytes. 

$ size ./branchless
       text    data     bss     dec     hex  filename
        446     117      48     611     263  branchless

Here's the first segment of disassembly of the binary.


    401000:  mov    eax,0x1
    401005:  mov    edi,0x1
    40100a:  movabs rsi,0x402000
    401014:  mov    edx,0xa
    401019:  syscall        <--- sys_write to stdout "username: password: ...">
    40101b:  mov    eax,0x0
    401020:  mov    edi,0x0
    401025:  movabs rsi,0x402078
    40102f:  mov    edx,0x8
    401034:  syscall        <--- sys_read from stdin, to read username>
    401036:  mov    eax,0x1
    40103b:  mov    edi,0x1
    401040:  movabs rsi,0x40200a
    40104a:  mov    edx,0xa
    40104f:  syscall        <--- sys_write asking for password>
    401051:  mov    eax,0x0
    401056:  mov    edi,0x0
    40105b:  movabs rsi,0x402080
    401065:  mov    edx,0x11
    40106a:  syscall        <--- sys_read to get password input>

The username is stored at 0x402078 and the password at 0x402080.

Following that we have some form of a function initialization:


    40106c:  push   rbp
    40106d:  mov    rbp,rsp
    401070:  sub    rsp,0x20
    401074:  mov    QWORD PTR [rbp-0x8],0x4010e8
    40107c:  mov    QWORD PTR [rbp-0x10],0x40109e
    401084:  mov    QWORD PTR [rbp-0x18],0x401000
    40108c:  xor    rax,rax
    40108f:  xor    rbx,rbx
    401092:  xor    rcx,rcx
    401095:  xor    rdx,rdx

A stack frame is created. Registers are being cleared. We have the start of .text initialized to rbp - 0x18. Two more instruction addresses are stored at rbp - 0x10 = 0x40109e and at rbp - 0x8 = 0x4010e8.

The following seems to be the code that does the jumping calculations. Here's where the branching is done. 


    401098:  mov    r10d,0x8
    40109e:  mov    rax,rdx
    4010a1:  imul   rax,rax,0x7
    4010a5:  mov    bl,BYTE PTR [rdx+0x402078]
    4010ab:  add    rax,rbx
    4010ae:  and    rax,0x1f
    4010b2:  mov    cl,BYTE PTR [rax+0x402055]
    4010b8:  mov    BYTE PTR [rdx+0x402091],cl
    4010be:  imul   rbx,rcx
    4010c2:  and    rbx,0x1f
    4010c6:  mov    cl,BYTE PTR [rbx+0x402055]
    4010cc:  mov    BYTE PTR [r10+0x402091],cl
    4010d3:  inc    rdx
    4010d6:  inc    r10
    4010d9:  cmp    rdx,0x8
    4010dd:  mov    rax,QWORD PTR [rbp-0x10]
    4010e1:  cmove  rax,QWORD PTR [rbp-0x8]
    4010e6:  jmp    rax
